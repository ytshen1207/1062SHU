## 透過計算字元數的方式僅擷取最後四碼年份
youtuber = youtuber %>%
  mutate(n_char = nchar(as.character(CREATED))) %>%
  mutate(YEAR = as.numeric(substr(CREATED, n_char-3, n_char)))

## 儲存整理完畢的資料，以後就可以直接讀取使用
saveRDS(youtuber, "20180326_files/youtuber.rds")
