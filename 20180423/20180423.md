``` r

library(dplyr)

df = youtuber %>%
  group_by(COUNTRY) %>%
  summarise(subs = mean(SUBSCRIBERS)) %>%
  joinCountryData2Map("ISO2", "COUNTRY")

mapCountryData(df, "subs")
```
